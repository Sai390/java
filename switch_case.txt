
class Main
{
public static void main(String args[])
{
char color='a';

switch(color)
{
case 'a':
System.out.println("blue");
break;
case 'c':
System.out.println("red");
break;
case 'b':
System.out.println("black");
break;
case'd':
System.out.println("yellow");
break;
case 'e':
System.out.println("orange");
break;
default:
System.out.println("white");
}
}
}
